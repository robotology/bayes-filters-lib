94 template<typename Derived, typename DerivedScalar = typename Derived::Scalar, bfl::utils::enable_if_t<std::is_floating_point<DerivedScalar>::value, bool> = true>
95 Eigen::Matrix<DerivedScalar, 3, Eigen::Dynamic> quaternion_to_rotation_vector(const Eigen::MatrixBase<Derived>& quaternion)
96 {
97  Eigen::Matrix<DerivedScalar, 3, Eigen::Dynamic> rotation_vectors(3, quaternion.cols());
98 
99  for (std::size_t i = 0; i < quaternion.cols(); ++i)
100  {
101  const DerivedScalar norm_n = quaternion.col(i).tail(3).norm();
102  if (norm_n > 1e-4)
103  {
104  const DerivedScalar w = quaternion.col(i)(0);
105  if (w < 0)
106  rotation_vectors.col(i) = - 2.0 * std::acos(-w) * quaternion.col(i).tail(3) / norm_n;
107  else
108  rotation_vectors.col(i) = 2.0 * std::acos(w) * quaternion.col(i).tail(3) / norm_n;
109  }
110  else
111  rotation_vectors.col(i) = Eigen::Matrix<DerivedScalar, 3, 1>::Zero();
112  }
113 
114  return rotation_vectors;
115 }

131 template<typename Derived, typename DerivedScalar = typename Derived::Scalar, bfl::utils::enable_if_t<std::is_floating_point<DerivedScalar>::value, bool> = true>
132 Eigen::Matrix<DerivedScalar, 4, Eigen::Dynamic> rotation_vector_to_quaternion(const Eigen::MatrixBase<Derived>& rotation_vector)
133 {
134  Eigen::Matrix<DerivedScalar, 4, Eigen::Dynamic> quaternions(4, rotation_vector.cols());
135  for (std::size_t i = 0; i < rotation_vector.cols(); ++i)
136  {
137  const DerivedScalar norm_r = rotation_vector.col(i).norm();
138  if (norm_r > 1e-4)
139  {
140  quaternions.col(i)(0) = std::cos(norm_r / 2.0);
141  quaternions.col(i).tail(3) = std::sin(norm_r / 2.0) * rotation_vector.col(i) / norm_r;

143  else
144  {
145  quaternions.col(i)(0) = 1.0;
146  quaternions.col(i).tail(3) = Eigen::Matrix<DerivedScalar, 3, 1>::Zero();
147  }
148  }

168 template<typename Derived, typename DerivedScalar = typename Derived::Scalar, bfl::utils::enable_if_t<std::is_floating_point<DerivedScalar>::value, bool> = true>
169 Eigen::Matrix<DerivedScalar, 4, Eigen::Dynamic> sum_quaternion_rotation_vector(const Eigen::MatrixBase<Derived>& quaternion, const Eigen::MatrixBase<Derived>& rotation_vector)
170 {
171  /* Move rotation vectors to their quaternionic representation. */
172  Eigen::Matrix<DerivedScalar, 4, Eigen::Dynamic> vector_as_quaternion = rotation_vector_to_quaternion(rotation_vector);
173 
174  Eigen::Quaternion<DerivedScalar> q_right(quaternion.col(0)(0), quaternion.col(0)(1), quaternion.col(0)(2), quaternion.col(0)(3));
175 
176  Eigen::Matrix<DerivedScalar, 4, Eigen::Dynamic> quaternions(4, rotation_vector.cols());
177  for (std::size_t i = 0; i < rotation_vector.cols(); ++i)
178  {
179  Eigen::Quaternion<DerivedScalar> q_left(vector_as_quaternion.col(i)(0), vector_as_quaternion.col(i)(1), vector_as_quaternion.col(i)(2), vector_as_quaternion.col(i)(3));
180  Eigen::Quaternion<DerivedScalar> sum = q_left * q_right;
181

183  quaternions.col(i).tail(3) = sum.vec();

204 template<typename Derived, typename DerivedScalar = typename Derived::Scalar, bfl::utils::enable_if_t<std::is_floating_point<DerivedScalar>::value, bool> = true>
205 Eigen::Matrix<DerivedScalar, 3, Eigen::Dynamic> diff_quaternion(const Eigen::MatrixBase<Derived>& quaternion_left, const Eigen::MatrixBase<Derived>& quaternion_right)
206 {
207  Eigen::Matrix<DerivedScalar, 4, Eigen::Dynamic> products(4, quaternion_left.cols());
208 
209  Eigen::Quaternion<DerivedScalar> q_right(quaternion_right.col(0)(0), quaternion_right.col(0)(1), quaternion_right.col(0)(2), quaternion_right.col(0)(3));
210  Eigen::Quaternion<DerivedScalar> q_right_conj = q_right.conjugate();
211 
212  /* Products between each left quaternion and the conjugated right quaternion. */
213  for (std::size_t i = 0; i < quaternion_left.cols(); ++i)
214  {
215  Eigen::Quaternion<DerivedScalar> q_left(quaternion_left.col(i)(0), quaternion_left.col(i)(1), quaternion_left.col(i)(2), quaternion_left.col(i)(3));
216  Eigen::Quaternion<DerivedScalar> product = q_left * q_right_conj;
217 
218  products.col(i)(0) = product.w();
219  products.col(i).tail(3) = product.vec();
220  }
221 
222  /* Express displacements in the tangent space. */
223  return quaternion_to_rotation_vector(products);
224 }

240 template<typename Derived, typename DerivedScalar = typename Derived::Scalar, bfl::utils::enable_if_t<std::is_floating_point<DerivedScalar>::value, bool> = true>
241 Eigen::Matrix<DerivedScalar, 4, 1> mean_quaternion(const Eigen::MatrixBase<Derived>& weight, const Eigen::MatrixBase<Derived>& quaternion)
242 {
243  /* Weighted outer product of quaternions. */
244  Eigen::Matrix<DerivedScalar, 4, 4> outer_product_mean = Eigen::Matrix<DerivedScalar, 4, 4>::Zero();
245  for (std::size_t i = 0; i < weight.rows(); ++i)
246  outer_product_mean.noalias() += weight.col(0)(i) * quaternion.col(i) * quaternion.col(i).transpose();
247 
248  /* Take the weighted mean as the eigenvector corresponding to the maximum eigenvalue. */
249  Eigen::EigenSolver<Eigen::Matrix<DerivedScalar, 4, 4>> eigen_solver(outer_product_mean);
250  Eigen::Matrix<std::complex<DerivedScalar>, 4, 1> eigenvalues(eigen_solver.eigenvalues());
251  int maximum_index;
252  eigenvalues.real().maxCoeff(&maximum_index);
253  return eigen_solver.eigenvectors().real().block(0, maximum_index, 4, 1);
254 }

300 template<typename Derived>
301 Eigen::VectorXd multivariate_gaussian_log_density_UVR(const Eigen::MatrixBase<Derived>& input, const Eigen::Ref<const Eigen::VectorXd>& mean, const Eigen::Ref<const Eigen::MatrixXd>& U, const Eigen::Ref<const Eigen::MatrixXd>& V, const Eigen::Ref<const Eigen::MatrixXd>& R)
302 {
303  std::size_t input_size = input.rows();
304  std::size_t block_size = R.rows();
305  std::size_t num_blocks = input_size / block_size;
306 
307  const auto diff = input.colwise() - mean;
308 
309  /* Evaluate inv(R) */
310  Eigen::MatrixXd inv_R(block_size, input_size);
311  if (R.cols() == block_size)
312  {
313  // The diagonal blocks of R are all equal
314  Eigen::MatrixXd inv_R_single = R.inverse();
315  for (std::size_t i = 0; i < num_blocks; i++)
316  {
317  inv_R.block(0, block_size * i, block_size, block_size) = inv_R_single;
318  }
319  }
320  else
321  {
322  // The diagonal blocks of R are different
323  for (std::size_t i = 0; i < num_blocks; i++)
324  {
325  inv_R.block(0, block_size * i, block_size, block_size) = R.block(0, block_size * i, block_size, block_size).inverse();
326  }
327  }
328 
329  /* Evaluate V * inv(R) */
330  Eigen::MatrixXd V_inv_R(V.rows(), V.cols());
331  for (std::size_t i = 0; i < V.cols() / block_size; i++)
332  {
333  V_inv_R.middleCols(i * block_size, block_size) = V.middleCols(i * block_size, block_size) * inv_R.block(0, block_size * i, block_size, block_size);
334  }
335 
336  /* Evaluate diff^{T} * inv(R) */
337  Eigen::MatrixXd diff_T_inv_R(input.cols(), input.rows());
338  for (std::size_t i = 0; i < num_blocks; i++)
339  {
340  diff_T_inv_R.middleCols(i * block_size, block_size) = diff.middleRows(i * block_size, block_size).transpose() * inv_R.block(0, block_size * i, block_size, block_size);
341  }
342 
343  /* Evaluate I + V * inv(R) * U */
344  Eigen::MatrixXd I_V_inv_R_U = Eigen::MatrixXd::Identity(V.rows(), V.rows()) + V_inv_R * U;
345 
351  Eigen::VectorXd weighted_diffs(input.cols());
352  for (std::size_t i = 0; i < weighted_diffs.size(); i++)
353  weighted_diffs(i) = diff_T_inv_R.row(i) * (Eigen::MatrixXd::Identity(U.rows(), U.rows()) - U * I_V_inv_R_U.inverse() * V_inv_R) * diff.col(i);
354 
360  double det_S;
361  double det_R = 1.0;
362  if (R.cols() == block_size)
363  det_R = std::pow(R.determinant(), num_blocks);
364  else
365  for (std::size_t i = 0; i < num_blocks; i++)
366  det_R *= R.block(0, block_size * i, block_size, block_size).determinant();
367 
368  det_S = det_R * I_V_inv_R_U.determinant();
369 
370  /* Evaluate the full logarithm density */
371  Eigen::VectorXd values(input.cols());
372  for (std::size_t i = 0; i < input.cols(); i++)
373  values(i) = - 0.5 * (static_cast<double>(diff.rows()) * std::log(2.0 * M_PI) + std::log(det_S) + weighted_diffs(i));
374 
375  return values;
376 }
