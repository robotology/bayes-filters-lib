23 ) :

25  simulated_state_model_(std::move(simulated_state_model))
26 {
27  /* Since a LinearSensor is intended as a sensor that measure
28  * the state directly (i.e. no linear combination of the states),
29  * then it is possible to extract the output size from the
30  * output size of the simulated state model.
31  */
32  std::size_t dim_linear_state;
33  std::tie(dim_linear_state, std::ignore) = simulated_state_model_->getStateModel().getOutputSize();
34 
35  dim_linear_ = 0;
36  dim_circular_ = 0;
37 
38  for (std::size_t i = 0; i < H_.rows(); ++i)
39  {
40  Eigen::MatrixXf::Index state_index;
41  H_.row(i).array().abs().maxCoeff(&state_index);
42 
43  if (state_index < dim_linear_state)
44  dim_linear_++;
45  else
46  dim_circular_++;
47  }
48 }
