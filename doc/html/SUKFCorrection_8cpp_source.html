48 std::pair<bool, VectorXd> SUKFCorrection::getLikelihood()
49 {
50  /* This method uses the member variable propagated_sigma_points_.
51  After each call to correctStep(), that variable contains a matrix Y
52  such that the predicted covariance of measurement is S = Y * Y^{T} + R */
53 
54  if ((innovations_.rows() == 0) || (innovations_.cols() == 0))
55  return std::make_pair(false, VectorXd());
56 
57  /* Evaluate the matrix containing the measurement noise covariance for each submeasurement. */
58  MatrixXd R(measurement_sub_size_, innovations_.rows());
59  for (std::size_t i = 0; i < innovations_.rows() / measurement_sub_size_; i++)
60  {
61  R.middleCols(i * measurement_sub_size_, measurement_sub_size_) = getNoiseCovarianceMatrix(i);
62  }
63 
64  /* Evaluate the likelihoods. */
65  VectorXd likelihood(innovations_.cols());
66  std::size_t size_sigma_points = propagated_sigma_points_.cols() / innovations_.cols();
67  for (std::size_t i = 0; i < innovations_.cols(); i++)
68  {
69  Ref<MatrixXd> Y = propagated_sigma_points_.middleCols(size_sigma_points * i, size_sigma_points);
70  likelihood(i) = utils::multivariate_gaussian_density_UVR(innovations_.col(i), VectorXd::Zero(innovations_.rows()), Y, Y.transpose(), R).coeff(0);
71  }
72 
73  return std::make_pair(true, likelihood);
74 }

77 void SUKFCorrection::correctStep(const GaussianMixture& pred_state, GaussianMixture& corr_state)
78 {
79  /* Get the current measurement if available. */
80  bool valid_measurement;
81  Data measurement;
82  std::tie(valid_measurement, measurement) = measurement_model_->measure();
83 
84  VectorDescription meas_description = measurement_model_->getMeasurementDescription();
85 
86  /* Check if the size of the measurement is compatible with measurement_sub_size. */
87  std::size_t meas_size = meas_description.total_size();
88  valid_measurement &= (((meas_size) % measurement_sub_size_ ) == 0);
89 
90  if (!valid_measurement)
91  {
92  corr_state = pred_state;
93  return;
94  }
95 
96  /* Sample sigma points. */
97  MatrixXd input_sigma_points = sigma_point::sigma_point(pred_state, ut_weight_.c);
98 
99  /* Propagate sigma points. */
100  Data pred;
101  bool valid_pred;
102  std::tie(valid_pred, pred) = measurement_model_->predictedMeasure(input_sigma_points);
103 
104  if (!valid_pred)
105  {
106  corr_state = pred_state;
107  return;
108  }
109 
110  /* Cast data to MatrixXd. */
111  propagated_sigma_points_ = bfl::any::any_cast<MatrixXd&&>(std::move(pred));
112 
113  /* Evaluate the predicted mean. */
114  std::size_t size_sigmas = (pred_state.dim * 2) + 1;
115  MatrixXd pred_mean(meas_size, pred_state.components);
116  for (size_t i = 0; i < pred_state.components; i++)
117  {
118  Ref<MatrixXd> prop_sp = propagated_sigma_points_.middleCols(size_sigmas * i, size_sigmas);
119 
120  /* Evaluate the mean. */
121  pred_mean.col(i).noalias() = prop_sp * ut_weight_.mean;
122  }
123 
124  /* Evaluate the innovation if possible. */
125  bool valid_innovation;
126  Data innovation;
127  std::tie(valid_innovation, innovation) = measurement_model_->innovation(pred_mean, measurement);
128 
129  if (!valid_innovation)
130  {
131  corr_state = pred_state;
132  return;
133  }
134 
135  /* Cast innovations once for all. */
136  innovations_ = any::any_cast<MatrixXd&&>(std::move(innovation));
137 
138  /* From now on using equations from the paper:
139  Barfoot, T., McManus, C. (2011),
140  'A Serial Approach to Handling High-Dimensional Measurements in the Sigma-Point Kalman Filter.',
141  Science and Systems VII,
142  MIT Press */
143 
144  /* Process all the components in the mixture. */
145  MatrixXd sqrt_ut_weight = ut_weight_.covariance.array().sqrt().matrix().asDiagonal();
146  for (size_t i = 0; i < pred_state.components; i++)
147  {
148  /* Compose square root of the measurement covariance matrix.
149  IV.C.2.b */
150  Ref<MatrixXd> Y = propagated_sigma_points_.middleCols(size_sigmas * i, size_sigmas);
151 
152  /* Shift w.r.t. the mean. */
153  Y.colwise() -= pred_mean.col(i);
154 
155  /* Weight using square root of unscented transform weight. */
156  Y *= sqrt_ut_weight;
157 
158  /* Compose matrix C and vector d by cycling over all the sub-vector of the measurement
159  IV.C.2.c */
160  MatrixXd C_inv = MatrixXd::Identity(size_sigmas, size_sigmas);
161  VectorXd d = VectorXd::Zero(size_sigmas);
162 
163  for (size_t j = 0; j < (meas_size) / measurement_sub_size_; j++)
164  {
165  MatrixXd tmp(size_sigmas, measurement_sub_size_);
166  tmp.noalias() = Y.middleRows(measurement_sub_size_ *j, measurement_sub_size_).transpose() * getNoiseCovarianceMatrix(j).inverse();
167 
168  C_inv += tmp * Y.middleRows(measurement_sub_size_ *j, measurement_sub_size_);
169 
170  d += tmp * innovations_.col(i).middleRows(measurement_sub_size_ *j, measurement_sub_size_);
171  }
172 
173  /* Process input sigma points.
174  IV.C.3 */
175  Ref<MatrixXd> X = input_sigma_points.middleCols(size_sigmas * i, size_sigmas);
176  X.topRows(pred_state.dim_linear).colwise() -= pred_state.mean(i).topRows(pred_state.dim_linear);
177  X.bottomRows(pred_state.dim_circular) = directional_sub(X.bottomRows(pred_state.dim_circular), pred_state.mean(i).bottomRows(pred_state.dim_circular));
178  X *= sqrt_ut_weight;
179 
180  C_inv = C_inv.inverse();
181 
182  /* Evaluate the filtered mean.
183  IV.C.4 */
184  corr_state.mean(i) = pred_state.mean(i) + X * C_inv * d;
185 
186  /* Evaluate the filtered covariance.
187  IV.C.4 */
188  corr_state.covariance(i) = X * C_inv * X.transpose();
189  }
190 }

193 MatrixXd SUKFCorrection::getNoiseCovarianceMatrix(const std::size_t index)
194 {
195  /* Obtain the noise covariance matrix from the measurement model. */
196  MatrixXd R;
197  std::tie(std::ignore, R) = measurement_model_->getNoiseCovarianceMatrix();
198 
199  if (use_reduced_noise_covariance_matrix_)
200  return R;
201  else
202  return R.block(measurement_sub_size_ * index, measurement_sub_size_ * index, measurement_sub_size_, measurement_sub_size_);
